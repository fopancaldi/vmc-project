48template <Dimension D, ParticNum N>
49Energy Mean(std::vector<LocEnAndPoss<D, N>> const &v) {
50 assert(v.size() > 1);
51 auto const size = std::ssize(v);
52
53 return std::accumulate(v.begin(), v.end(), Energy{0},
54 [](Energy e, LocEnAndPoss<D, N> leps) { return e + leps.localEn; }) /
55 static_cast<FPType>(size);
56};

63template <Dimension D, ParticNum N>
64Energy StdDev(std::vector<LocEnAndPoss<D, N>> const &v) {
65 assert(v.size() > 1);
66 auto const size = std::ssize(v);
67
68 Energy const mean = Mean(v);
69 EnSquared const meanVar = std::accumulate(v.begin(), v.end(), EnSquared{0},
70 [mean](EnSquared es, LocEnAndPoss<D, N> const &leps) {
71 return es + (leps.localEn - mean) * (leps.localEn - mean);
72 }) /
73 static_cast<FPType>(size * (size - 1));
74 return sqrt(meanVar);
75}

82template <Dimension D, ParticNum N>
83Energy GetStat(std::vector<LocEnAndPoss<D, N>> const &v, Statistic stat) {
84 Energy result;
85 switch (stat) {
86 case Statistic::mean:
87 result = Mean(v);
88 break;
89 case Statistic::stdDev:
90 result = StdDev(v);
91 break;
92 default:
93 assert(false);
94 }
95 return result;

107template <Dimension D, ParticNum N>
108std::vector<LocEnAndPoss<D, N>> GetStatOfEachBlock(std::vector<LocEnAndPoss<D, N>> const &energies,
109 IntType blockSize, IntType numBlocks, Statistic stat) {
110 assert(numBlocks > 0);
111 IntType const numEnergies = static_cast<IntType>(std::ssize(energies));
112 assert(numEnergies > 0);
113 assert((numEnergies % blockSize) == 0);
114
115 // The vector which will contain the statistic of each block
116 std::vector<LocEnAndPoss<D, N>> blockStats;
117 blockStats.reserve(static_cast<long unsigned int>(numBlocks));
118
119 std::generate_n(std::back_inserter(blockStats), numBlocks,
120 [&energies, &numEnergies, &blockSize, &stat, currentBlock = IntType{0}]() mutable {
121 auto start = energies.begin() + currentBlock * blockSize;
122 auto end = start + blockSize;
123
124 LocEnAndPoss<D, N> blockLEPs;
125 blockLEPs.localEn = GetStat(std::vector<LocEnAndPoss<D, N>>(start, end), stat);
126 Position<D> fakePosition;
127 std::fill(fakePosition.begin(), fakePosition.end(),
128 Coordinate{std::numeric_limits<FPType>::quiet_NaN()});
129 std::fill(blockLEPs.positions.begin(), blockLEPs.positions.end(), fakePosition);
130
131 ++currentBlock;
132 return blockLEPs;
133 });
134 return blockStats;
135}

145template <Dimension D, ParticNum N>
146std::vector<std::vector<LocEnAndPoss<D, N>>> BootstrapSamples(std::vector<LocEnAndPoss<D, N>> const &energies,
147 IntType boostrapSamples, RandomGenerator &gen) {
148 IntType const numEnergies = static_cast<IntType>(std::ssize(energies));
149 assert(numEnergies > 0);
150 assert(boostrapSamples > 0);
151
152 std::uniform_int_distribution<> dist(0, numEnergies - 1);
153
154 std::vector<std::vector<LocEnAndPoss<D, N>>> bootstrapSamples;
155 bootstrapSamples.reserve(static_cast<unsigned long int>(boostrapSamples));
156
157 // Resample with replacement
158 std::generate_n(std::back_inserter(bootstrapSamples), boostrapSamples,
159 [&energies, &numEnergies, &dist, &gen]() {
160 std::vector<LocEnAndPoss<D, N>> sample;
161 sample.reserve(static_cast<unsigned long int>(numEnergies));
162
163 // Fill the current sample with random energies
164 std::generate_n(std::back_inserter(sample), numEnergies, [&]() {
165 LocEnAndPoss<D, N> result = energies[static_cast<UIntType>(dist(gen))];
166 Position<D> fakePosition;
167 std::fill(fakePosition.begin(), fakePosition.end(),
168 Coordinate{std::numeric_limits<FPType>::quiet_NaN()});
169 std::fill(result.positions.begin(), result.positions.end(), fakePosition);
170 return result;
171 });
172
173 return sample;
174 });
175 return bootstrapSamples;
176}

186template <Dimension D, ParticNum N>
187std::vector<LocEnAndPoss<D, N>>
188BootstrapLEPs(std::vector<std::vector<LocEnAndPoss<D, N>>> const &bootstrapSamples, IntType boostrapSamples,
189 Statistic stat) {
190 std::vector<LocEnAndPoss<D, N>> bootstrapVector;
191 bootstrapVector.reserve(static_cast<long unsigned int>(boostrapSamples));
192
193 std::generate_n(std::back_inserter(bootstrapVector), boostrapSamples,
194 [&bootstrapSamples, &stat, currentSample = UIntType{0u}]() mutable {
195 const auto &sample = bootstrapSamples[currentSample];
196 ++currentSample;
197
198 LocEnAndPoss<D, N> leps;
199 leps.localEn = GetStat(sample, stat);
200 Position<D> fakePosition;
201 std::fill(fakePosition.begin(), fakePosition.end(),
202 Coordinate{std::numeric_limits<FPType>::quiet_NaN()});
203 std::fill(leps.positions.begin(), leps.positions.end(), fakePosition);
204 return leps;
205 });
206 return bootstrapVector;
207}

224template <Dimension D, ParticNum N>
225BlockingResult EvalBlocking(std::vector<LocEnAndPoss<D, N>> const &energies) {
226 IntType const numEnergies = static_cast<IntType>(std::ssize(energies));
227 assert(numEnergies > 0);
228
229 std::vector<IntType> blockSizes;
230 std::vector<Energy> means;
231 std::vector<Energy> stdDevs;
232
233 unsigned long int const reservedSize = static_cast<unsigned long int>(std::log2(numEnergies) - 1);
234 blockSizes.reserve(reservedSize);
235 means.reserve(reservedSize);
236 stdDevs.reserve(reservedSize);
237
238 for (IntType blockSize = 2; blockSize <= numEnergies / 2; blockSize *= 2) {
239 IntType numBlocks = static_cast<IntType>(static_cast<FPType>(numEnergies) / blockSize);
240
241 std::vector<LocEnAndPoss<D, N>> blockMeans =
242 GetStatOfEachBlock(energies, blockSize, numBlocks, Statistic::mean);
243 blockSizes.push_back(blockSize);
244
245 // Statistics
246 Energy meanOfMeans = GetStat(blockMeans, Statistic::mean);
247 means.push_back(meanOfMeans);
248 Energy stdDevOfMeans = GetStat(blockMeans, Statistic::stdDev);
249 stdDevs.push_back(stdDevOfMeans);
250 }
251 assert(blockSizes.size() == means.size());
252 assert(blockSizes.size() == stdDevs.size());
253 return BlockingResult{blockSizes, means, stdDevs};
254}

266template <Dimension D, ParticNum N>
267Energy BlockingAnalysis(std::vector<LocEnAndPoss<D, N>> const &energies) {
268 IntType const numEnergies = static_cast<IntType>(std::ssize(energies));
269 // Check that numEnergies is a power of 2 using bitwise AND between the number n and (n - 1)
270 assert((numEnergies & (numEnergies - 1)) == 0);
271 assert(numEnergies > 1);
272
273 BlockingResult blockingResult = EvalBlocking(energies);
274
275 // Find the first pair of elements where the difference is below the threshold
276 auto pltIt =
277 std::adjacent_find(blockingResult.stdDevs.begin(), blockingResult.stdDevs.end(),
278 [](Energy e1, Energy e2) { return abs(e2 - e1) < threshold_blockingAnalysis; });
279 assert(pltIt != blockingResult.stdDevs.end());
280 Energy bestStdDev = *pltIt;
281
282 return bestStdDev;
283}

291template <Dimension D, ParticNum N>
292Energy BootstrapAnalysis(std::vector<LocEnAndPoss<D, N>> const &energies, IntType const &boostrapSamples,
293 RandomGenerator &gen) {
294 // Generate sample with replacement
295 std::vector<std::vector<LocEnAndPoss<D, N>>> bootstrapSamples =
296 BootstrapSamples(energies, boostrapSamples, gen);
297
298 // Calculate std. dev. of each sample vector and place into bootstrapStdDevs
299 std::vector<LocEnAndPoss<D, N>> bootstrapStdDevs =
300 BootstrapLEPs(bootstrapSamples, boostrapSamples, Statistic::stdDev);
301
302 Energy stdDev = GetStat(bootstrapStdDevs, Statistic::mean);
303 return stdDev;
304}

321template <Dimension D, ParticNum N>
322Energy ErrorOnAvg(std::vector<LocEnAndPoss<D, N>> const &energies, StatFuncType function,
323 IntType const &boostrapSamples, RandomGenerator &gen) {
324 switch (function) {

329 return BootstrapAnalysis(energies, boostrapSamples, gen);

334 assert(false);
