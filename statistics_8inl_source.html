59template <Dimension D, ParticNum N>
60Energy Mean(std::vector<LocEnAndPoss<D, N>> const &v) {
61 assert(v.size() > 1);
62 auto const size = std::ssize(v);
63
64 return std::accumulate(v.begin(), v.end(), Energy{0},
65 [](Energy e, LocEnAndPoss<D, N> leps) { return e + leps.localEn; }) /
66 static_cast<FPType>(size);
67};

74template <Dimension D, ParticNum N>
75Energy StdDev(std::vector<LocEnAndPoss<D, N>> const &v) {
76 assert(v.size() > 1);
77 auto const size = std::ssize(v);
78
79 Energy const mean = Mean(v);
80 EnSquared const meanVar = std::accumulate(v.begin(), v.end(), EnSquared{0},
81 [mean](EnSquared es, LocEnAndPoss<D, N> const &leps) {
82 return es + (leps.localEn - mean) * (leps.localEn - mean);
83 }) /
84 static_cast<FPType>(size * (size - 1));
85 return sqrt(meanVar);
86}

240template <Dimension D, ParticNum N>
241BlockingResult EvalBlocking(std::vector<LocEnAndPoss<D, N>> const &energies, IntType const &numEnergies) {
242 assert(numEnergies > 0);
243
244 std::vector<IntType> blockSizes;
245 std::vector<Energy> means;
246 std::vector<Energy> stdDevs;
247
248 unsigned long int const reservedSize = static_cast<unsigned long int>(std::log2(numEnergies) - 1);
249 blockSizes.reserve(reservedSize);
250 means.reserve(reservedSize);
251 stdDevs.reserve(reservedSize);
252
253 for (IntType blockSize = 2; blockSize <= numEnergies / 2; blockSize *= 2) {
254 IntType numBlocks = static_cast<IntType>(static_cast<FPType>(numEnergies) / blockSize);
255
256 std::vector<LocEnAndPoss<D, N>> blockMeans =
257 EvalStatBlocks(energies, numEnergies, blockSize, numBlocks, Statistic::mean);
258 blockSizes.push_back(blockSize);
259
260 // Statistics
261 Energy meanOfMeans = GetStat(blockMeans, Statistic::mean);
262 means.push_back(meanOfMeans);
263 Energy stdDevOfMeans = GetStat(blockMeans, Statistic::stdDev);
264 stdDevs.push_back(stdDevOfMeans);
265 }
266 assert(blockSizes.size() == means.size());
267 assert(blockSizes.size() == stdDevs.size());
268 return BlockingResult{blockSizes, means, stdDevs};
269}

283template <Dimension D, ParticNum N>
284Energy BlockingAnalysis(std::vector<LocEnAndPoss<D, N>> const &energies) {
285 IntType const numEnergies = static_cast<IntType>(std::ssize(energies));
286 // Check that numEnergies is a power of 2 using bitwise AND between the number n and (n - 1)
287 assert((numEnergies & (numEnergies - 1)) == 0);
288 assert(numEnergies > 1);
289
290 BlockingResult blockingResult = EvalBlocking(energies, numEnergies);
291
292 // Find the first pair of elements where the difference is below the threshold
293 auto pltIt =
294 std::adjacent_find(blockingResult.stdDevs.begin(), blockingResult.stdDevs.end(),
295 [](Energy e1, Energy e2) { return abs(e2 - e1) < threshold_blockingAnalysis; });
296 assert(pltIt != blockingResult.stdDevs.end());
297 Energy bestStdDev = *pltIt;
298
299 return bestStdDev;
300}

308template <Dimension D, ParticNum N>
309Energy BootstrapAnalysis(std::vector<LocEnAndPoss<D, N>> const &energies, IntType const &numSamples,
310 RandomGenerator &gen) {
311 IntType const numEnergies = static_cast<IntType>(std::ssize(energies));
312 assert(numEnergies > 1);
313 std::uniform_int_distribution<> dist(0, numEnergies - 1);
314
315 // Generate sample with replacement
316 std::vector<std::vector<LocEnAndPoss<D, N>>> bootstrapSamples =
317 BootstrapSamples(energies, numEnergies, numSamples, gen, dist);
318
319 // Calculate std. dev. of each sample vector and place into bootstrapStdDevs
320 std::vector<LocEnAndPoss<D, N>> bootstrapStdDevs =
321 BootstrapLEPs(bootstrapSamples, numSamples, Statistic::stdDev);
322
323 Energy stdDev = GetStat(bootstrapStdDevs, Statistic::mean);
324 return stdDev;
325}

343template <Dimension D, ParticNum N>
344Energy Statistics(std::vector<LocEnAndPoss<D, N>> const &energies, StatFuncType function,
345 IntType const &numSamples, RandomGenerator &gen) {
346 switch (function) {

352 break;
