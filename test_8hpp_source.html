68template <vmcp::Dimension D, vmcp::ParticNum N, vmcp::VarParNum V, class Wavefunction, class Potential>
69vmcp::Positions<D, N> FindPeak_(Wavefunction const &wavef, vmcp::VarParams<V> params, Potential const &pot,
70 vmcp::CoordBounds<D> bounds, vmcp::IntType numPoints,
71 vmcp::RandomGenerator &gen) {
72 using namespace vmcp;
73
74 static_assert(IsWavefunction<D, N, V, Wavefunction>());
75 static_assert(IsPotential<D, N, Potential>());
76 assert(numPoints > 0);
77
78 Position<D> center;
79 std::transform(bounds.begin(), bounds.end(), center.begin(),
80 [](Bound<Coordinate> b) { return (b.upper + b.lower) / 2; });
81 Positions<D, N> result;
82 std::fill(result.begin(), result.end(), center);
83 std::uniform_real_distribution<FPType> unif(0, 1);
84 std::mutex m;
85 auto const indices = std::ranges::views::iota(0, numPoints);
86 std::for_each(std::execution::par, indices.begin(), indices.end(), [&](int) {
87 Positions<D, N> newPoss;
88 for (Position<D> &p : newPoss) {
89 std::transform(bounds.begin(), bounds.end(), p.begin(), [&unif, &gen](Bound<Coordinate> b) {
90 return b.lower + (b.upper - b.lower) * unif(gen);
91 });
92 }
93 // The requirement ... > minPsi avoids having wavef(...) = nan in the future, which breaks the update
94 // algorithms
95 {
96 std::lock_guard<std::mutex> l(m);
97 if ((pot(newPoss) > pot(result)) && (wavef(newPoss, params) > minWavef_peakSearch)) {
98 result = newPoss;
99 }
100 }
101 });
102 return result;
103}
