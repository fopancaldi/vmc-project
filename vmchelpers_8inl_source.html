54template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class Potential>
55Positions<D, N> FindPeak_(Wavefunction const &wavef, VarParams<V> params, Potential const &pot,
56 CoordBounds<D> bounds, IntType numPoints, RandomGenerator &gen) {
57 static_assert(IsWavefunction<D, N, V, Wavefunction>());
58 static_assert(IsPotential<D, N, Potential>());
59 assert(numPoints > 0);
60
61 Position<D> center;
62 std::transform(bounds.begin(), bounds.end(), center.begin(),
63 [](Bound<Coordinate> b) { return (b.upper + b.lower) / 2; });
64 // FP TODO: Can you make it more elegant?
65 Positions<D, N> result;
66 std::fill(result.begin(), result.end(), center);
67 std::uniform_real_distribution<FPType> unif(0, 1);
68 std::mutex m;
69 auto const indices = std::ranges::views::iota(0, numPoints);
70 // FP TODO: Data race in unif(gen)
71 // Maybe put another mutex? Deadlock risk?
72 std::for_each(std::execution::par, indices.begin(), indices.end(), [&](int) {
73 Positions<D, N> newPoss;
74 for (Position<D> &p : newPoss) {
75 std::transform(bounds.begin(), bounds.end(), p.begin(), [&unif, &gen](Bound<Coordinate> b) {
76 return b.lower + (b.upper - b.lower) * unif(gen);
77 });
78 }
79 // The requirement ... > minPsi avoids having wavef(...) = nan in the future, which breaks the update
80 // algorithms
81 {
82 std::lock_guard<std::mutex> l(m);
83 if ((pot(newPoss) > pot(result)) && (wavef(newPoss, params) > minWavef_peakSearch)) {
84 result = newPoss;
85 }
86 }
87 });
88 return result;
89}

101template <Dimension D, ParticNum N, VarParNum V, class FirstDerivative, class Wavefunction>
102std::array<std::array<FPType, D>, N> DriftForceAnalytic_(Wavefunction const &wavef, Positions<D, N> poss,
103 VarParams<V> params,
104 Gradients<D, N, FirstDerivative> const &grads) {
105 static_assert(IsWavefunction<D, N, V, Wavefunction>());
106 static_assert(IsWavefunctionDerivative<D, N, V, FirstDerivative>());
107
108 std::array<std::array<FPType, D>, N> gradsVal;
109 std::array<std::array<FPType, D>, N> result;
110 for (ParticNum n = 0u; n != N; ++n) {
111 std::transform(grads[n].begin(), grads[n].end(), gradsVal[n].begin(),
112 [&poss, params](FirstDerivative const &fd) { return fd(poss, params); });
113 std::transform(gradsVal[n].begin(), gradsVal[n].end(), result[n].begin(),
114 [&wavef, &poss, params](FPType f) { return 2 * f / wavef(poss, params); });
115 }
116
117 return result;
118}

152template <Dimension D, ParticNum N, VarParNum V, class Wavefunction>
153IntType MetropolisUpdate_(Wavefunction const &wavef, VarParams<V> params, Positions<D, N> &poss, FPType step,
154 RandomGenerator &gen) {
155 static_assert(IsWavefunction<D, N, V, Wavefunction>());
156
157 IntType succesfulUpdates = 0;
158 for (Position<D> &p : poss) {
159 Position const oldPos = p;
160 FPType const oldPsi = wavef(poss, params);
161 std::uniform_real_distribution<FPType> unif(0, 1);
162 std::transform(p.begin(), p.end(), p.begin(), [&gen, &unif, step](Coordinate c) {
163 // FP TODO: Convert step to Coordinate?
164 return c + Coordinate{(unif(gen) - FPType{0.5f}) * step};
165 });
166 if (unif(gen) < std::pow(wavef(poss, params) / oldPsi, 2)) {
167 ++succesfulUpdates;
168 } else {
169 p = oldPos;
170 }
171 }
172 return succesfulUpdates;
173}

180template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class FirstDerivative>
181IntType ImportanceSamplingUpdate_(Wavefunction const &wavef, VarParams<V> params,
182 Gradients<D, N, FirstDerivative> const &grads, Masses<N> masses,
183 Positions<D, N> &poss, RandomGenerator &gen) {
184 static_assert(IsWavefunction<D, N, V, Wavefunction>());
185 static_assert(IsWavefunctionDerivative<D, N, V, FirstDerivative>());
186
187 std::array<FPType, N> diffConsts;
188 std::transform(masses.begin(), masses.end(), diffConsts.begin(),
189 [](Mass m) { return hbar * hbar / (2 * m.val); });
190
191 IntType successfulUpdates = 0;
192 for (ParticNum n = 0u; n != N; ++n) {
193 Position<D> &p = poss[n];
194 Position const oldPos = p;
195 FPType const oldPsi = wavef(poss, params);
196 std::array<std::array<FPType, D>, N> const oldDriftForce =
197 DriftForceAnalytic_<D, N, V>(wavef, poss, params, grads);
198 std::normal_distribution<FPType> normal(0, diffConsts[n] * deltaT);
199 for (Dimension d = 0u; d != D; ++d) {
200 p[d].val = oldPos[d].val + diffConsts[n] * deltaT * (oldDriftForce[n][d] + normal(gen));
201 }
202 FPType const newPsi = wavef(poss, params);
203
204 FPType forwardExponent = 0;
205 for (Dimension d = 0u; d != D; ++d) {
206 forwardExponent -=
207 std::pow(p[d].val - oldPos[d].val - diffConsts[n] * deltaT * oldDriftForce[n][d], 2) /
208 (4 * diffConsts[n] * deltaT);
209 }
210 FPType const forwardProb = std::exp(forwardExponent);
211
212 std::array<std::array<FPType, D>, N> const newDriftForce =
213 DriftForceAnalytic_<D, N, V>(wavef, poss, params, grads);
214 FPType backwardExponent = 0;
215 for (Dimension d = 0u; d != D; ++d) {
216 backwardExponent -=
217 std::pow(oldPos[d].val - p[d].val - diffConsts[n] * deltaT * newDriftForce[n][d], 2) /
218 (4 * diffConsts[n] * deltaT);
219 }
220 FPType const backwardProb = std::exp(backwardExponent);
221
222 FPType const acceptanceRatio = (newPsi * newPsi * backwardProb) / (oldPsi * oldPsi * forwardProb);
223 std::uniform_real_distribution<FPType> unif(0, 1);
224 if (unif(gen) < acceptanceRatio) {
225 ++successfulUpdates;
226 } else {
227 p = oldPos;
228 }
229 }
230 return successfulUpdates;
231}

273template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class Laplacian, class Potential>
274Energy LocalEnergyAnalytic_(Wavefunction const &wavef, VarParams<V> params,
275 Laplacians<N, Laplacian> const &lapls, Masses<N> masses, Potential const &pot,
276 Positions<D, N> poss) {
277 static_assert(IsWavefunction<D, N, V, Wavefunction>());
278 static_assert(IsWavefunctionDerivative<D, N, V, Laplacian>());
279 static_assert(IsPotential<D, N, Potential>());
280
281 FPType weightedLaplSum = 0;
282 // FP TODO: Bad indices, bad!
283 for (ParticNum n = 0u; n != N; ++n) {
284 weightedLaplSum += lapls[n](poss, params) / masses[n].val;
285 }
286 return Energy{-(hbar * hbar / 2) * (weightedLaplSum / wavef(poss, params)) + pot(poss)};
287}

297template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class Potential>
298Energy LocalEnergyNumeric_(Wavefunction const &wavef, VarParams<V> params, FPType derivativeStep,
299 Masses<N> masses, Potential const &pot, Positions<D, N> poss) {
300 static_assert(IsWavefunction<D, N, V, Wavefunction>());
301 static_assert(IsPotential<D, N, Potential>());
302
303 Energy result{pot(poss)};
304 // FP TODO: Can you parallelize this?
305 for (ParticNum n = 0u; n != N; ++n) {
306 for (Dimension d = 0u; d != D; ++d) {
307 result.val += -std::pow(hbar, 2) / (2 * masses[n].val) *
308 (8 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{5 * derivativeStep}), params) -
309 125 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{4 * derivativeStep}), params) +
310 1000 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{3 * derivativeStep}), params) -
311 6000 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{2 * derivativeStep}), params) +
312 42000 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{derivativeStep}), params) -
313 73766 * wavef(poss, params) +
314 42000 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-derivativeStep}), params) -
315 6000 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-2 * derivativeStep}), params) +

336template <Dimension D, ParticNum N, VarParNum V, class Wavefunction>
337std::array<Energy, V> ReweightedEnergies_(Wavefunction const &wavef, VarParams<V> oldParams,
338 std::vector<LocEnAndPoss<D, N>> oldLEPs, FPType step) {
339 static_assert(IsWavefunction<D, N, V, Wavefunction>());
340
341 // FP TODO: can you use std::transform here?
342 std::array<Energy, V> result;
343 for (UIntType v = 0u; v != V; ++v) {
344 VarParams<V> newParams = oldParams;
345 newParams[v].val += step;
346 std::vector<Energy> reweightedLocalEnergies(oldLEPs.size());
347 std::transform(
348 std::execution::par_unseq, oldLEPs.begin(), oldLEPs.end(), reweightedLocalEnergies.begin(),
349 [&wavef, newParams, oldParams](LocEnAndPoss<D, N> const &lep) {
350 return Energy{std::pow(wavef(lep.positions, newParams) / wavef(lep.positions, oldParams), 2) *
351 lep.localEn.val};
352 });
353
354 FPType const numerator =
355 std::accumulate(reweightedLocalEnergies.begin(), reweightedLocalEnergies.end(), FPType{0},
356 [](FPType f, Energy e) { return f + e.val; });
357 FPType const denominator = std::accumulate(
358 oldLEPs.begin(), oldLEPs.end(), FPType{0},
359 [&wavef, newParams, oldParams](FPType f, LocEnAndPoss<D, N> const &lep) {
360 return f + std::pow(wavef(lep.positions, newParams) / wavef(lep.positions, oldParams), 2);
361 });
362
363 result[v] = Energy{numerator / denominator};
364 }
365 return result;
366}
