45template <Dimension D, ParticNum N, VarParNum V, class FirstDerivative, class Wavefunction>
46std::array<std::array<FPType, D>, N> DriftForceAnalytic_(Wavefunction const &wavef, Positions<D, N> poss,
47 VarParams<V> params,
48 Gradients<D, N, FirstDerivative> const &grads) {
49 static_assert(IsWavefunction<D, N, V, Wavefunction>());
50 static_assert(IsWavefunctionDerivative<D, N, V, FirstDerivative>());
51
52 std::array<std::array<FPType, D>, N> result;
53
54 std::transform(std::execution::par_unseq, grads.begin(), grads.end(), result.begin(),
55 [&wavef, &poss, params](Gradient<D, FirstDerivative> const &g) {
56 std::array<FPType, D> result_;
57 std::transform(g.begin(), g.end(), result_.begin(),
58 [&wavef, &poss, params](FirstDerivative const &fd) {
59 return 2 * fd(poss, params) / wavef(poss, params);
60 });
61 return result_;
62 });
63
64 return result;
65}

73template <Dimension D, ParticNum N, VarParNum V, class Wavefunction>
74std::array<std::array<FPType, D>, N> DriftForceNumeric_(Wavefunction const &wavef, Positions<D, N> poss,
75 VarParams<V> params, FPType step) {
76 static_assert(IsWavefunction<D, N, V, Wavefunction>());
77
78 std::array<std::array<FPType, D>, N> result;
79 auto const indices = std::ranges::views::iota(VarParNum{0u}, N);
80 std::for_each(
81 std::execution::par_unseq, indices.begin(), indices.end(),
82 [&wavef, &poss, params, step, &result](ParticNum n) {
83 for (Dimension d = 0u; d != D; ++d) {
84 // Numerical derivative correct up to O(step^9)
85 result[n][d] =
86 2 *
87 (FPType{1} / 280 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-4 * step}), params) +
88 FPType{-4} / 105 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-3 * step}), params) +
89 FPType{1} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-2 * step}), params) +
90 FPType{-4} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-step}), params) +
91 FPType{4} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{step}), params) +
92 FPType{-1} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{2 * step}), params) +
93 FPType{4} / 105 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{3 * step}), params) +
94 FPType{-1} / 280 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{4 * step}), params)) /
95 (step * wavef(poss, params));
96 }
97 });
98
99 return result;
100}

115template <Dimension D, ParticNum N, VarParNum V, class Wavefunction>
116IntType MetropolisUpdate_(Wavefunction const &wavef, VarParams<V> params, Positions<D, N> &poss, FPType step,
117 RandomGenerator &gen) {
118 static_assert(IsWavefunction<D, N, V, Wavefunction>());
119 assert(wavef(poss, params) > 1e-12);
120
121 IntType succesfulUpdates = 0;
122 for (Position<D> &p : poss) {
123 Position const oldPos = p;
124 FPType const oldPsi = wavef(poss, params);
125 std::uniform_real_distribution<FPType> unif(0, 1);
126 std::transform(p.begin(), p.end(), p.begin(), [&gen, &unif, step](Coordinate c) {
127 return c + Coordinate{(unif(gen) - FPType{0.5f}) * step};
128 });
129 if (unif(gen) < std::pow(wavef(poss, params) / oldPsi, 2)) {
130 ++succesfulUpdates;
131 } else {
132 p = oldPos;
133 }
134 }
135 return succesfulUpdates;
136}

155template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class FirstDerivative>
156IntType ImportanceSamplingUpdate_(Wavefunction const &wavef, VarParams<V> params, bool useAnalytical,
157 FPType derivativeStep, Gradients<D, N, FirstDerivative> const &grads,
158 Masses<N> masses, Positions<D, N> &poss, RandomGenerator &gen) {
159 static_assert(IsWavefunction<D, N, V, Wavefunction>());
160 static_assert(IsWavefunctionDerivative<D, N, V, FirstDerivative>());
161
162 std::array<FPType, N> diffConsts;
163 std::transform(masses.begin(), masses.end(), diffConsts.begin(),
164 [](Mass m) { return hbar * hbar / (2 * m.val); });
165
166 IntType successfulUpdates = 0;
167 for (ParticNum n = 0u; n != N; ++n) {
168 Position<D> &p = poss[n];
169 Position const oldPos = p;
170 FPType const oldPsi = wavef(poss, params);
171
172 std::array<std::array<FPType, D>, N> oldDriftForce;
173 if (useAnalytical) {
174 oldDriftForce = DriftForceAnalytic_<D, N, V>(wavef, poss, params, grads);
175 } else {
176 oldDriftForce = DriftForceNumeric_<D, N, V>(wavef, poss, params, derivativeStep);
177 }
178
179 // Jensen in his notes, section 1.4.3, suggests a value between 0.001 and 0.01
180 FPType const timeStep = 0.005;
181
182 std::normal_distribution<FPType> normal(0, 1);
183
184 for (Dimension d = 0u; d != D; ++d) {
185 p[d].val = oldPos[d].val + diffConsts[n] * timeStep * oldDriftForce[n][d] +
186 normal(gen) * std::sqrt(timeStep);
187 }
188
189 FPType const newPsi = wavef(poss, params);
190
191 FPType forwardExponent = 0;
192 for (Dimension d = 0u; d != D; ++d) {
193 forwardExponent -=
194 std::pow(p[d].val - oldPos[d].val - diffConsts[n] * timeStep * oldDriftForce[n][d], 2) /
195 (4 * diffConsts[n] * timeStep);
196 }
197 FPType const forwardProb = std::exp(forwardExponent);
198
199 std::array<std::array<FPType, D>, N> newDriftForce;
200 if (useAnalytical) {
201 newDriftForce = DriftForceAnalytic_<D, N, V>(wavef, poss, params, grads);
202 } else {
203 newDriftForce = DriftForceNumeric_<D, N, V>(wavef, poss, params, derivativeStep);
204 }
205 FPType backwardExponent = 0;
206 for (Dimension d = 0u; d != D; ++d) {
207 backwardExponent -=
208 std::pow(oldPos[d].val - p[d].val - diffConsts[n] * timeStep * newDriftForce[n][d], 2) /
209 (4 * diffConsts[n] * timeStep);
210 }
211 FPType const backwardProb = std::exp(backwardExponent);
212
213 FPType const acceptanceRatio = (newPsi * newPsi * backwardProb) / (oldPsi * oldPsi * forwardProb);
214 std::uniform_real_distribution<FPType> unif(0, 1);
215 if (unif(gen) < acceptanceRatio) {
216 ++successfulUpdates;
217 } else {
218 p = oldPos;
219 }
220 }
221 return successfulUpdates;
222}

264template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class Laplacian, class Potential>
265Energy LocalEnergyAnalytic_(Wavefunction const &wavef, VarParams<V> params,
266 Laplacians<N, Laplacian> const &lapls, Masses<N> masses, Potential const &pot,
267 Positions<D, N> poss) {
268 static_assert(IsWavefunction<D, N, V, Wavefunction>());
269 static_assert(IsWavefunctionDerivative<D, N, V, Laplacian>());
270 static_assert(IsPotential<D, N, Potential>());
271
272 FPType const weightedLaplSum =
273 std::inner_product(lapls.begin(), lapls.end(), masses.begin(), FPType{0}, std::plus<>(),
274 [&poss, params](Laplacian const &l, Mass m) { return l(poss, params) / m.val; });
275 return Energy{-hbar * hbar * weightedLaplSum / (2 * wavef(poss, params)) + pot(poss)};
276}

286template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class Potential>
287Energy LocalEnergyNumeric_(Wavefunction const &wavef, VarParams<V> params, FPType step, Masses<N> masses,
288 Potential const &pot, Positions<D, N> poss) {
289 static_assert(IsWavefunction<D, N, V, Wavefunction>());
290 static_assert(IsPotential<D, N, Potential>());
291
292 Energy result{pot(poss)};
293 std::mutex m;
294 auto const indices = std::ranges::views::iota(VarParNum{0u}, N);
295 std::for_each(std::execution::par_unseq, indices.begin(), indices.end(), [&](ParticNum n) {
296 for (Dimension d = 0u; d != D; ++d) {
297 // Numerical derivative correct up to O(step^9)
298 Energy const temp =
299 Energy{-hbar * hbar / (2 * masses[n].val) *
300 (FPType{-1} / 560 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-4 * step}), params) +
301 FPType{8} / 315 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-3 * step}), params) +
302 FPType{-1} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-2 * step}), params) +
303 FPType{8} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{-step}), params) +
304 FPType{-205} / 72 * wavef(poss, params) +
305 FPType{8} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{step}), params) +
306 FPType{-1} / 5 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{2 * step}), params) +
307 FPType{8} / 315 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{3 * step}), params) +
308 FPType{-1} / 560 * wavef(MoveBy_<D, N>(poss, d, n, Coordinate{4 * step}), params)) /
309 (std::pow(step, 2) * wavef(poss, params))};
310
311 {
312 std::lock_guard<std::mutex> l(m);
313 result += temp;
314 }
315 }
316 });
317
318 return result;
319}

332template <Dimension D, ParticNum N, VarParNum V, class Wavefunction>
333std::array<Energy, V> ReweightedEnergies_(Wavefunction const &wavef, VarParams<V> oldParams,
334 std::vector<LocEnAndPoss<D, N>> oldLEPs, FPType step) {
335 static_assert(IsWavefunction<D, N, V, Wavefunction>());
336
337 std::array<Energy, V> result;
338 std::generate_n(result.begin(), V, [&, v = VarParNum{0u}]() mutable {
339 VarParams<V> newParams = oldParams;
340 newParams[v] += VarParam{step};
341 std::vector<Energy> reweightedLocEns(oldLEPs.size());
342 std::transform(
343 std::execution::par_unseq, oldLEPs.begin(), oldLEPs.end(), reweightedLocEns.begin(),
344 [&wavef, newParams, oldParams](LocEnAndPoss<D, N> const &lep) {
345 return Energy{std::pow(wavef(lep.positions, newParams) / wavef(lep.positions, oldParams), 2) *
346 lep.localEn.val};
347 });
348 std::vector<FPType> denomAddends(oldLEPs.size());
349 std::transform(std::execution::par_unseq, oldLEPs.begin(), oldLEPs.end(), denomAddends.begin(),
350 [&wavef, newParams, oldParams](LocEnAndPoss<D, N> const &lep) {
351 return std::pow(wavef(lep.positions, newParams) / wavef(lep.positions, oldParams),
352 2);
353 });
354
355 Energy const num = std::reduce(std::execution::par_unseq, reweightedLocEns.begin(),
356 reweightedLocEns.end(), Energy{0}, std::plus<>());
357 FPType const den = std::reduce(std::execution::par_unseq, denomAddends.begin(), denomAddends.end(),
358 FPType{0}, std::plus<>());
359
360 ++v;
361 return num / den;
362 });
363 return result;
364}
