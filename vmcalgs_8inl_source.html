53template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class FirstDerivative, class Laplacian,
54 class Potential>
55std::vector<LocEnAndPoss<D, N>>
56VMCLocEnAndPoss_(Wavefunction const &wavef, Positions<D, N> poss, VarParams<V> params, bool useAnalytical,
57 bool useImpSamp, Gradients<D, N, FirstDerivative> const &grads,
58 Laplacians<N, Laplacian> const &lapls, FPType derivativeStep, Masses<N> masses,
59 Potential const &pot, CoordBounds<D> bounds, IntType numEnergies, RandomGenerator &gen) {
60 static_assert(IsWavefunction<D, N, V, Wavefunction>());
61 static_assert(IsWavefunctionDerivative<D, N, V, FirstDerivative>());
62 static_assert(IsWavefunctionDerivative<D, N, V, Laplacian>());
63 static_assert(IsPotential<D, N, Potential>());
64 assert(numEnergies > 0);
65
66 // Choose the initial step
67 Bound const smallestBound =
68 *(std::min_element(bounds.begin(), bounds.end(), [](Bound<Coordinate> b1, Bound<Coordinate> b2) {
69 return b1.Length().val < b2.Length().val;
70 }));
71 FPType step = smallestBound.Length().val / stepDenom_vmcLEPs;
72
73 std::function<Energy()> localEnergy;
74 if (useAnalytical) {
75 localEnergy = std::function<Energy()>{
76 [&]() { return LocalEnergyAnalytic_<D, N>(wavef, params, lapls, masses, pot, poss); }};
77 } else {
78 localEnergy = std::function<Energy()>{
79 [&]() { return LocalEnergyNumeric_<D, N>(wavef, params, derivativeStep, masses, pot, poss); }};
80 }
81 std::function<IntType()> update;
82 if (useImpSamp) {
83 update = std::function<IntType()>{[&]() {
84 return ImportanceSamplingUpdate_<D, N>(wavef, params, useAnalytical, derivativeStep, grads,
85 masses, poss, gen);
86 }};
87 } else {
88 update = std::function<IntType()>{
89 [&]() { return MetropolisUpdate_<D, N>(wavef, params, poss, step, gen); }};
90 }
91
92 std::vector<LocEnAndPoss<D, N>> result;
93 result.reserve(static_cast<long unsigned int>(numEnergies));
94 // Move away from the starting ponit, in order to forget the dependence on the initial conditions
95 for (IntType i = 0; i != movesForgetICs_vmcLEPs; ++i) {
96 update();
97 }
98 for (IntType i = 0; i != numEnergies; ++i) {
99 IntType succesfulUpdates = 0;
100 for (IntType j = 0; j != autocorrelationMoves_vmcLEPs; ++j) {
101 succesfulUpdates += update();
102 }
103 result.emplace_back(localEnergy(), poss);
104
105 // Adjust the step size
106 // Call car = current acc. rate, tar = target acc. rate
107 // Add (car - tar)/tar to step, since it increases step if too many moves were accepted and decreases
108 // it if too few were accepted
109 FPType currentAcceptRate = succesfulUpdates * FPType{1} / (autocorrelationMoves_vmcLEPs * N);
110 step *= (currentAcceptRate > targetAcceptRate_vmcLEPs ? FPType{11} / 10 : FPType{9} / 10);
111 }
112
113 return result;
114}

128template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class LocEnAndPossCalculator>
129VMCResult<V> VMCRBestParams_(VarParams<V> initialParams, ParamBounds<V> bounds, Wavefunction const &wavef,
130 LocEnAndPossCalculator const &lepsCalc, StatFuncType function,
131 IntType const &boostrapSamples, RandomGenerator &gen) {
132 static_assert(IsWavefunction<D, N, V, Wavefunction>());
133 static_assert(
134 std::is_invocable_r_v<std::vector<LocEnAndPoss<D, N>>, LocEnAndPossCalculator, VarParams<V>>);
135 static_assert(V != 0);
136 assert(!std::isnan(initialParams[0].val));
137
138 // Use a gradient descent algorithm with termination condition: stop if the next step is too small
139 // compared to the current parameters
140 VMCResult<V> result;
141 Energy currentEn;
142 VarParams<V> currentParams = initialParams;
143 FPType const initialParamsNorm = std::sqrt(
144 std::inner_product(initialParams.begin(), initialParams.end(), initialParams.begin(), FPType{0},
145 std::plus<>(), [](VarParam v1, VarParam v2) { return v1.val * v2.val; }));
146 FPType gradStep = initialParamsNorm / stepDenom_gradDesc;
147 std::array<FPType, V> oldMomentum;
148 std::fill_n(oldMomentum.begin(), V, FPType{0});
149
150 for (IntType i = 0; i != maxLoops_gradDesc; ++i) {
151 // The gradient descent should end in a reasonable time
152 assert((i + 1) != maxLoops_gradDesc);
153 // Better to be turned off if numWalkers_gradDesc != 1
154 std::cout << "Variational Parameter: " << currentParams[0].val << "\n";
155
156 // Update the energy
157 std::vector<LocEnAndPoss<D, N>> const currentLEPs = lepsCalc(currentParams);
158 currentEn = Mean(currentLEPs);
159
160 // Compute the gradient by using reweighting and update the momentum
161 std::array<Energy, V> energiesIncreasedParam =
162 ReweightedEnergies_<D, N, V>(wavef, currentParams, currentLEPs, gradStep);
163 std::array<Energy, V> energiesDecreasedParam =
164 ReweightedEnergies_<D, N, V>(wavef, currentParams, currentLEPs, -gradStep);
165 std::array<FPType, V> currentMomentum;
166 std::generate_n(currentMomentum.begin(), V,
167 [v = VarParNum{0}, &energiesIncreasedParam, &energiesDecreasedParam, &oldMomentum,
168 gradStep]() mutable {
169 FPType const result_ =
170 -FPType{3} / 4 *
171 (energiesIncreasedParam[v].val - energiesDecreasedParam[v].val) /
172 (2 * gradStep) +
173 FPType{1} / 4 * oldMomentum[v];
174 assert(!std::isnan(result_));
175 ++v;
176 return result_;
177 });
178
179 // Set as next step used to compute the gradient the current gradient norm, which is also the size of
180 // the step if that step is accepted
181 FPType const currentParamsNorm = std::sqrt(
182 std::inner_product(currentParams.begin(), currentParams.end(), currentParams.begin(), FPType{0},
183 std::plus<>(), [](VarParam v1, VarParam v2) { return v1.val * v2.val; }));
184 gradStep = std::sqrt(std::inner_product(currentMomentum.begin(), currentMomentum.end(),
185 currentMomentum.begin(), FPType{0}));
186
187 // Check the termination condition
188 if (gradStep / currentParamsNorm < stoppingThreshold_gradDesc) {
189 result.energy = currentEn;
190 result.stdDev = ErrorOnAvg(currentLEPs, function, boostrapSamples, gen);
191 result.bestParams = currentParams;
192 break;
193 } else {
194 for (VarParNum v = 0u; v != V; ++v) {
195 FPType multiplier = 0.02f;
196 while ((currentParams[v].val + multiplier * currentMomentum[v] > bounds[v].upper.val) ||
197 (currentParams[v].val + multiplier * currentMomentum[v] < bounds[v].lower.val)) {
198 multiplier /= 2;
199 }
200 currentParams[v].val += multiplier * currentMomentum[v];
201 }
202 /*std::transform(currentParams.begin(), currentParams.end(), currentMomentum.begin(),
203 currentParams.begin(), [](VarParam cp, FPType cm) { return cp + VarParam{cm}; });*/
204 oldMomentum = currentMomentum;
205 }
206 }
207
208 return result;
209}

222template <Dimension D, ParticNum N, VarParNum V, class Wavefunction, class LocEnAndPossCalculator>
223VMCResult<V> VMCRBestParams_(ParamBounds<V> bounds, Wavefunction const &wavef,
224 LocEnAndPossCalculator const &lepsCalc, IntType numWalkers,
225 StatFuncType function, IntType const &boostrapSamples, RandomGenerator &gen) {
226 static_assert(IsWavefunction<D, N, V, Wavefunction>());
227 static_assert(
228 std::is_invocable_r_v<std::vector<LocEnAndPoss<D, N>>, LocEnAndPossCalculator, VarParams<V>>);
229 assert(numWalkers > IntType{0});
230
231 if constexpr (V == VarParNum{0}) {
232 VarParams<0u> const fakeParams{};
233 std::vector<LocEnAndPoss<D, N>> const vmcLEPs = lepsCalc(fakeParams);
234 return VMCResult<0>{Mean(vmcLEPs), ErrorOnAvg(vmcLEPs, function, boostrapSamples, gen),
235 VarParams<0>{}};
236 } else {
237 std::mutex m;
238 std::uniform_real_distribution<FPType> unif(0, 1);
239 std::vector<VMCResult<V>> vmcResults(static_cast<long unsigned int>(numWalkers));
240 std::generate_n(std::execution::par_unseq, vmcResults.begin(), numWalkers_gradDesc, [&]() {
241 unsigned long int seed_;
242 {
243 std::lock_guard<std::mutex> l(m);
244 seed_ = gen();
245 }
246 RandomGenerator localGen{seed_};
247 VarParams<V> initialParams;
248 for (VarParNum v = 0u; v != V; ++v) {
249 initialParams[v] = bounds[v].lower + bounds[v].Length() * unif(localGen);
250 }
251 return VMCRBestParams_<D, N, V>(initialParams, bounds, wavef, lepsCalc, function, boostrapSamples,
252 localGen);
253 });
254
255 return *std::min_element(
256 vmcResults.begin(), vmcResults.end(),
257 [](VMCResult<V> vmcr1, VMCResult<V> vmcr2) { return vmcr1.energy < vmcr2.energy; });
258 }
259}
