35template <Dimension D>
36CoordBounds<D> MakeCoordBounds(Coordinate lower, Coordinate upper) {
37 assert(lower.val < upper.val);
38 CoordBounds<D> coordBounds;
39 std::fill(coordBounds.begin(), coordBounds.end(), Bound{lower, upper});
40 assert(coordBounds.size() == D);
41 return coordBounds;
42}

48template <Dimension D>
49FPType Distance(Position<D> x, Position<D> y) {
50 FPType sqrdDist =
51 std::inner_product(x.begin(), x.end(), y.begin(), FPType{0}, std::plus<>(),
52 [](Coordinate &a, Coordinate &b) { return (a.val - b.val) * (a.val - b.val); });
53 return std::sqrt(sqrdDist);
54}

60template <Dimension D, ParticNum N>
61Positions<D, N> BuildFCCStartPoint_(CoordBounds<D> bounds, FPType latticeSpacing) {
62 static_assert(D == Dimension{3u});
63 Positions<D, N> result;
64
65 // Calculate the center of the bounds
66 Position<D> center;
67 for (Dimension d = 0; d < D; ++d) {
68 center[d].val = (bounds[d].lower.val + bounds[d].upper.val) / 2;
69 }
70
71 FPType const diagSpacing = latticeSpacing / std::sqrt(vmcp::FPType{2});
72
73 // Offset positions for FCC lattice arrangement
74 FPType const offset[4][3] = {{0, 0, 0},
75 {diagSpacing, diagSpacing, 0},
76 {diagSpacing, 0, diagSpacing},
77 {0, diagSpacing, diagSpacing}};
78
79 // Priority queue for positions to fill
80 std::queue<Position<D>> positionsToFill;
81 positionsToFill.push(center);
82
83 UIntType particlesPlaced = 0;
84 std::set<std::array<FPType, D>> visited;
85
86 // Place particles iteratively
87 while (!positionsToFill.empty() && particlesPlaced < N) {
88 Position<D> basePosition = positionsToFill.front();
89 positionsToFill.pop();
90
91 for (UIntType offsetIdx = 0; offsetIdx < 4; ++offsetIdx) {
92 Position<D> shiftedPos = basePosition;
93 for (Dimension d = 0; d < D; ++d) {
94 shiftedPos[d].val += offset[offsetIdx][d];
95 }
96
97 // Check if the position is unique
98 std::array<FPType, D> posKey;
99 for (Dimension d = 0; d < D; ++d) {
100 posKey[d] = shiftedPos[d].val;
101 }
102 if (visited.count(posKey)) {
103 continue;
104 }
105 visited.insert(posKey);
106
107 // Store the position
108 result[particlesPlaced] = shiftedPos;
109 ++particlesPlaced;
110
111 if (particlesPlaced == N) {
112 return result;
113 }
114
115 // Add neighboring positions to the queue
116 for (Dimension d = 0; d < D; ++d) {
117 for (IntType sign = -1; sign <= 1; sign += 2) {
118 Position<D> neighbor = basePosition;
119 neighbor[d].val += sign * latticeSpacing;
120 positionsToFill.push(neighbor);
121 }
122 }
123 }
124 }
125 return result;
126}

143template <Dimension D, ParticNum N>
144sciplot::Canvas MakeGraphNoInt(std::vector<FPType> const &alphaVals, std::vector<FPType> const &energyVals,
145 std::vector<ConfInterval> const &confInts, std::vector<FPType> const &exactEns,
146 std::string method) {
147 // Manipulate confidence intervals
148 std::vector<FPType> confIntErr(confInts.size());
